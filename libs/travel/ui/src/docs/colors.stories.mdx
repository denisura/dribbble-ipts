import { Meta, ColorPalette, ColorItem } from '@storybook/blocks';
import { theme } from '@ipts/travel-ui';

<Meta title="Travel/Colors" />

export const colorNames = [
  'primary',
  'secondary',
  'background',
  'common',
  'error',
  'neutral',
  'grey',
  'info',
  'success',
  'text',
];

<ColorPalette>

    {Object.entries(theme.palette)
    .filter(([name])=> colorNames.includes(name))
    .map(([name, colors])=>(

        key={name}
        title={name}
        colors={colors}
    />
    ))}

</ColorPalette>
